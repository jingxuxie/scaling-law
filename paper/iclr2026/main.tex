\title{Visible Spectrum: When Adaptive Optimizers Change Scaling Laws}
\author{Jingxu Xie et al.}

\begin{abstract}
Scaling laws guide decisions about model size, data, and compute, but they usually treat the optimizer as a constant-factor detail.  We show that this misses a basic mechanism: even in Gaussian linear regression, coordinatewise adaptive optimizers change scaling exponents only when their second moments make the covariance spectrum visible.  We introduce a \emph{visible spectrum} exponent $\theta\in[0,1]$.  If the diagonal preconditioner learned by RMSProp/Adam is spectrally comparable to $(\Sigma^\theta+\rho I)^{-1/2}$, then
\[
    \lambda_i(P^{1/2}\Sigma P^{1/2})
    \asymp
    \lambda_i(\Sigma)(\lambda_i(\Sigma)^\theta+\rho)^{-1/2},
    \qquad
    q_{\rm eff}=\theta/2.
\]
Thus, for $\lambda_i\asymp i^{-a}$, the learned-mode count changes from the SGD rate $n^{1/a}$ to $n^{1/[a(1-\theta/2)]}$ before damping.  With source exponent $b$, this yields sharp risk and compute laws, including the compute exponent
\[
    \beta(\theta)=\frac{b-1}{\max\{a(1-\theta/2),b\}+1}.
\]
The theory separates Adam's components: second moments determine the spectral exponent; first-moment momentum changes constants and stability; AdamW weight decay imposes a learned-mode cap that must decay with compute to preserve the no-decay law.  Experiments with deterministic filters, stochastic training, coordinate rotations, random-feature/sketched covariances, compute-optimal sweeps, and learning-rate grids support the mechanism.  Adaptive optimizers change scaling laws through \emph{visible spectral information}, not through adaptivity alone.
\end{abstract}

\section{Introduction}
Scaling laws have become central to modern machine learning.  They forecast loss as a function of data, parameters, and compute, and they have shaped the design of large language models and other frontier systems \citep{hestness2017deep,kaplan2020scaling,henighan2020scaling,hoffmann2022training}.  Yet optimizer choice is usually treated as a tuning detail.  This is at odds with practice: frontier systems are typically trained with Adam or AdamW, and optimizer hyperparameters often affect performance across scales.

This paper asks: \emph{When can an adaptive optimizer change a scaling exponent, rather than merely a constant?}  We answer in power-law Gaussian linear regression, a setting that is minimal but retains approximation error, optimization bias, variance/effective dimension, power-law spectra, source conditions, compute allocation, and optimizer-induced preconditioning \citep{bartlett2020benign,hastie2022surprises,bordelon2020spectrum,canatar2021spectral,lin2024scaling}.

\paragraph{Main idea.} Coordinatewise adaptive methods do not directly observe covariance eigenvalues.  They observe coordinatewise gradient second moments.  For Gaussian squared loss, those second moments are proportional to the diagonal of the covariance in the optimizer coordinate system.  If optimizer coordinates are aligned with covariance eigendirections, Adam/RMSProp behaves like damped spectral preconditioning with exponent $q=1/2$.  If coordinates are globally mixed so that all coordinate variances are flat, the same coordinatewise rule is scalar up to constants and cannot change the spectral exponent.

\paragraph{Contributions.}
\begin{enumerate}[leftmargin=1.4em,itemsep=0.25em]
    \item We prove a visible-spectrum theorem: if $P\asymp(\Sigma^\theta+\rho I)^{-1/2}$, then $q_{\rm eff}=\theta/2$ and the preconditioned spectrum is determined.
    \item We derive sharp risk and compute-optimal laws, including a visibility phase transition.
    \item We decompose Adam/RMSProp/AdamW: second moments determine $q_{\rm eff}$, momentum changes constants and stability, and AdamW weight decay creates a learned-mode cap requiring compute-dependent decay.
    \item We prove alignment results: eigenbasis and band-limited features expose spectrum; flat, Haar, and isotropic Gaussian sketches hide it.
    \item We validate the theory with deterministic filters, stochastic training, learning-rate sweeps, visible-profile interpolation, source diagnostics, compute-optimal sweeps, AdamW decay schedules, and random-feature/sketched-coordinate experiments.
\end{enumerate}

\section{Setup}
Let $x\sim\mathcal N(0,H)$ and $y=\langle x,w^\star\rangle+\xi$, where $\E\xi=0$ and $\E\xi^2=\sigma^2$.  A trainable feature representation is $z=Sx$, with covariance $\Sigma=\E[zz^\top]=SHS^\top$.  We measure excess risk $R(w)-\sigma^2=\|w-w^\star\|_\Sigma^2$.

\begin{assumption}[Power-law spectrum]
\label{ass:spectrum}
For some $a>1$, the ordered eigenvalues of $\Sigma$ satisfy $\lambda_i(\Sigma)\asymp i^{-a}$.
\end{assumption}

Let $\Sigma v_i=\lambda_i v_i$ and define source energies $s_i=\lambda_i\langle v_i,w^\star\rangle^2$.
\begin{assumption}[Source condition]
\label{ass:source}
For some $b>1$, $s_i\asymp i^{-b}$.  For noncommuting preconditioners, the risk theorem assumes the analogous condition in the transformed basis.  Appendix~\ref{app:source} shows that this is automatic for exact spectral preconditioners and holds in band-averaged form for invariant band-limited decompositions.
\end{assumption}
We write $N$ for samples/updates, $M$ for model dimension, $\Neff=N/\log N$, and $n=\Neff\gamma$.

\section{Visible spectrum of coordinatewise adaptivity}
\begin{lemma}[Coordinate second moments]
\label{lem:grad-second}
Let $z\sim\mathcal N(0,\Sigma)$, $y=\langle z,w^\star\rangle+\xi$, and $e=w-w^\star$.  For $g=(\langle z,e\rangle-\xi)z$,
\[
    \E[g_j^2\mid e]=c_e\Sigma_{jj}+2(\Sigma e)_j^2,
    \qquad c_e=\sigma^2+\|e\|_\Sigma^2.
\]
Consequently, $c_e\Sigma_{jj}\le \E[g_j^2\mid e]\le3c_e\Sigma_{jj}$.
\end{lemma}
\begin{proof}
Let $U=\langle z,e\rangle-\xi$ and $Z_j=z_j$.  Conditionally on $e$, $(U,Z_j)$ is centered Gaussian with $\E U^2=c_e$, $\E Z_j^2=\Sigma_{jj}$, and $\E[UZ_j]=(\Sigma e)_j$.  Isserlis' identity gives $\E[U^2Z_j^2]=\E U^2\E Z_j^2+2\E[UZ_j]^2$.  The upper bound follows from $(\Sigma e)_j^2\le(e^\top\Sigma e)\Sigma_{jj}\le c_e\Sigma_{jj}$.
\end{proof}

\begin{definition}[Visible spectral exponent]
\label{def:visible}
A diagonal adaptive preconditioner $P\succ0$ has visible exponent $\theta\in[0,1]$ at damping $\rho\ge0$ if there are constants $0<c\le C<\infty$, independent of scale, such that
\begin{equation}
\label{eq:comparability}
    c(\Sigma^\theta+\rho I)^{-1/2}\pre P\pre C(\Sigma^\theta+\rho I)^{-1/2}.
\end{equation}
\end{definition}

\begin{theorem}[Visible-spectrum theorem]
\label{thm:visible-spectrum}
Assume \eqref{eq:comparability}.  Let $\tSig=P^{1/2}\Sigma P^{1/2}$.  Then
\[
    \lambda_i(\tSig)\asymp\lambda_i(\Sigma)(\lambda_i(\Sigma)^\theta+\rho)^{-1/2}.
\]
If $\lambda_i(\Sigma)\asymp i^{-a}$ and damping is inactive on the range of interest, then
\[
    \lambda_i(\tSig)\asymp i^{-a(1-\theta/2)},\qquad q_{\rm eff}=\theta/2.
\]
\end{theorem}
\begin{proof}
The nonzero eigenvalues of $P^{1/2}\Sigma P^{1/2}$ equal those of $\Sigma^{1/2}P\Sigma^{1/2}$.  By \eqref{eq:comparability}, $\Sigma^{1/2}P\Sigma^{1/2}$ is Loewner-sandwiched by constant multiples of $\Sigma^{1/2}(\Sigma^\theta+\rho I)^{-1/2}\Sigma^{1/2}$.  The latter commutes with $\Sigma$ and has eigenvalues $\lambda_i(\Sigma)(\lambda_i(\Sigma)^\theta+\rho)^{-1/2}$.  The min-max principle proves the result.
\end{proof}

\begin{corollary}[Learned-mode count]
\label{cor:learned-count}
Let $K_{\rho,\theta}(n)=\#\{i:\lambda_i(\tSig)\gtrsim n^{-1}\}$.  For $\theta>0$,
\[
    K_{\rho,\theta}(n)\asymp
    \begin{cases}
    n^{1/[a(1-\theta/2)]}, & n\lesssim \rho^{-(1/\theta-1/2)},\\[4pt]
    \rho^{-1/(2a)}n^{1/a}, & n\gtrsim \rho^{-(1/\theta-1/2)}.
    \end{cases}
\]
For $\theta=0$, $K_{\rho,0}(n)\asymp n^{1/a}$.
\end{corollary}

\section{Risk, compute, and optimizer components}
Let $\tmu_i=\lambda_i(\tSig)$ and let $\ts_i$ be the source energy in the transformed basis.  The exact filter form is
\begin{equation}
\label{eq:filters}
    \E R_M(w_N)-\sigma^2\asymp
    \sum_{i>M}\ts_i+\sum_{i\le M}\frac{\ts_i}{1+n\tmu_i}
    +\frac1{\Neff}\sum_{i\le M}\min\{1,n^2\tmu_i^2\}.
\end{equation}

\begin{theorem}[Visible-spectrum risk law]
\label{thm:risk}
Let $\alpha=a(1-\theta/2)$.  Assume the transformed source condition $\ts_i\asymp i^{-b}$ and $1<b<\alpha+1$.  Then, up to constants and logarithmic factors,
\[
    \E R_M(w_N)-\sigma^2\asymp
    M^{1-b}+K_{\rho,\theta}(n)^{1-b}
    +\frac{\min\{M,K_{\rho,\theta}(n)\}}{\Neff}.
\]
\end{theorem}

Let compute be $C\asymp MN$.  In the pre-damping regime, the sharp proxy is
\begin{equation}
\label{eq:compute-proxy}
    R(M,N,K)-\sigma^2\asymp M^{1-b}+K^{1-b}+K/N,
\end{equation}
subject to $K\le N^{1/\alpha}$.

\begin{theorem}[Compute-optimal visible scaling]
\label{thm:compute}
Let $m=\max\{\alpha,b\}$ and $\alpha=a(1-\theta/2)$.  Then
\[
    M_\star(C)\asymp C^{1/(m+1)},\qquad
    N_\star(C)\asymp C^{m/(m+1)},\qquad
    K_\star(C)\asymp C^{1/(m+1)}.
\]
Moreover, $R_\star(C)-\sigma^2\asymp C^{-(b-1)/(m+1)}$, so
\[
    \beta(\theta)=\frac{b-1}{\max\{a(1-\theta/2),b\}+1}.
\]
\end{theorem}

\begin{corollary}[Visibility phase transition]
\label{cor:phase}
Define $\theta_c=2(1-b/a)$.  When $0<\theta_c<1$,
\[
    \beta(\theta)=
    \begin{cases}
    \dfrac{b-1}{a(1-\theta/2)+1}, & 0\le\theta\le\theta_c,\\[8pt]
    \dfrac{b-1}{b+1}, & \theta\ge\theta_c.
    \end{cases}
\]
Thus increasing visible spectral information improves the compute exponent only until the variance-limited phase.
\end{corollary}

\paragraph{Adam components.} The second moment determines $P$ and hence $q_{\rm eff}$.  First-moment momentum changes constants and stability but not the learned-mode cutoff: for fixed effective eigenvalue $\mu_i$, the population momentum polynomial has a root $1-\gamma\mu_i+O((\gamma\mu_i)^2)$.  AdamW decoupled weight decay adds a shrinkage floor $\delta/(\mu_i+\delta)$, so learned modes require $\mu_i\gtrsim\max\{n^{-1},\delta\}$.

\begin{theorem}[Compute-dependent AdamW decay]
\label{thm:adamw}
Let $m=\max\{\alpha,b\}$ and $\delta(C)=C^{-s}$.  Under the weight-decay cap, the compute-risk exponent is
\[
    \beta_{\rm wd}(\theta,s)=
    \min\left\{\frac{(b-1)s}{\alpha},\frac{b-1}{m+1}\right\}.
\]
The no-decay compute law is recovered iff $s\ge\alpha/(m+1)$.  Fixed weight decay corresponds to $s=0$ and creates an asymptotic floor.
\end{theorem}

\section{When is the spectrum visible?}
\begin{proposition}[Aligned coordinates]
If optimizer coordinates are eigenvectors of $\Sigma$, then $P=(\Sigma+\rho I)^{-1/2}$ and $\theta=1$.
\end{proposition}
\begin{proposition}[Band-limited mixing]
Suppose coordinates are partitioned into spectral bands $B_\ell$ such that $\lambda_i\asymp\Lambda_\ell$ for $i\in B_\ell$, and the optimizer basis mixes coordinates only within each band.  Then $P\asymp(\Sigma+\rho I)^{-1/2}$ up to band constants, so $\theta=1$ and $q_{\rm eff}=1/2$.
\end{proposition}
\begin{proposition}[Flat and globally mixed coordinates]
If $d_{\min}\le\Sigma_{jj}\le d_{\max}$ with $d_{\max}/d_{\min}=O(1)$, then $P\asymp cI$ and the spectrum of $P^{1/2}\Sigma P^{1/2}$ is the spectrum of $\Sigma$ up to constants.  Hence $\theta=0$ and $q_{\rm eff}=0$.
\end{proposition}

\section{Experiments}
Our experiments isolate the mechanisms in the theory.  We report both final risk and $q_{\rm eff}$: risk measures finite-time performance, while $q_{\rm eff}$ identifies the scaling-law mechanism.

\paragraph{Deterministic filters and stochastic visibility.} Deterministic sweeps verify learned-count predictions for SGD, Adam/RMSProp-like spectra, and AdamW weight decay.  Mini-batch training confirms that RMSProp/Adam/AdamW learn $q_{\rm eff}\approx1/2$ in aligned and band-limited coordinates, while flat/Haar coordinates have $q_{\rm eff}\approx0$.

\paragraph{Visible-profile interpolation.} A controlled stochastic experiment uses $P_{\theta,i}=(\lambda_i^\theta+\rho)^{-1/2}$.  For $a=1.5,b=1.4$, tuned final risk decreases monotonically as $\theta$ increases (Table~\ref{tab:theta}).

\section{Related work}
\paragraph{Scaling laws.} Empirical scaling laws have been studied across language modeling, generative modeling, and transfer \citep{hestness2017deep,rosenfeld2019constructive,kaplan2020scaling,henighan2020scaling,hernandez2021scaling,hoffmann2022training}.  Theoretical accounts connect power laws to kernel spectra, task-model alignment, and high-dimensional regression \citep{bordelon2020spectrum,canatar2021spectral,bahri2021explaining,lin2024scaling}.  The closest baseline is \citet{lin2024scaling}, who prove model/data/compute scaling laws for one-pass SGD in sketched linear regression.

\paragraph{Adaptive optimization and preconditioning.} Adaptive and preconditioned methods include AdaGrad \citep{duchi2011adaptive}, RMSProp \citep{tieleman2012rmsprop}, Adam \citep{kingma2015adam}, AdamW \citep{loshchilov2019decoupled}, natural gradient \citep{amari1998natural}, K-FAC \citep{martens2015optimizing}, Shampoo \citep{gupta2018shampoo}, and large-scale second-order methods \citep{anil2020scalable}.  Prior theory often focuses on convergence, regret, or implicit bias \citep{reddi2018convergence,wilson2017marginal}.  We ask whether the preconditioner reshapes a power-law spectrum enough to change scaling exponents.

\paragraph{Random features, kernels, and high-dimensional regression.} Random features and kernel approximations provide tractable models of feature learning and generalization \citep{rahimi2007random,rudi2017generalization,bach2017breaking,jacot2018neural}.  Power-law eigenspectra also appear in kernel learning curves and double descent \citep{belkin2019reconciling,hastie2022surprises,bartlett2020benign}.  Our results show that the same covariance spectrum can lead to different optimizer exponents depending on whether the feature coordinates expose the spectrum to coordinatewise second moments.

\paragraph{Optimizer-dependent scaling.} Empirical work has emphasized that optimizer choice and hyperparameter transfer affect observed scaling behavior \citep{mccandlish2018empirical,smith2018bayesian,ramani2026optimizer}.  Our contribution is a theorem-level mechanism: the visible spectrum controls the exponent.

\section{Limitations and scope}
The theory is for Gaussian linear regression and feature-coordinate models.  Neural networks have non-Gaussian gradients, changing representations, normalization layers, and interactions between optimization and representation learning.  Our results should be read as a mechanistic scaling-law model, not a complete theory of deep learning.  Finite-time risk improvements can occur even when $q_{\rm eff}\approx0$; our claim concerns spectral scaling exponents, not all optimizer advantages.

\section{Conclusion}
Adaptive optimizers do not automatically change scaling laws.  They change scaling exponents when their coordinatewise second moments make the covariance spectrum visible.  The visible exponent $\theta$ determines $q_{\rm eff}=\theta/2$, which determines the learned-mode count, risk law, compute-optimal allocation, and AdamW decay schedule.

\section{Proof details}
\subsection{Source stability}\label{app:source}
If $P=f(\Sigma)$ is spectral, then $\tSig=P^{1/2}\Sigma P^{1/2}=\Sigma f(\Sigma)$ and $u^\star=P^{-1/2}w^\star$.  For each eigenvector $v_i$,
\[
    \ts_i=\lambda_i f(\lambda_i)\left(f(\lambda_i)^{-1/2}\langle v_i,w^\star\rangle\right)^2=s_i.
\]
For invariant band decompositions, total source mass is preserved blockwise:
\[
    (P_\ell^{-1/2}w^\star_\ell)^\top(P_\ell^{1/2}\Sigma_\ell P_\ell^{1/2})(P_\ell^{-1/2}w^\star_\ell)=(w^\star_\ell)^\top\Sigma_\ell w^\star_\ell.
\]

\subsection{Risk filters}
For two-slope spectra, integral comparison gives
\[
    \sum_{i\le M}\frac{\ts_i}{1+n\tmu_i}\asymp K(n)^{1-b}
\]
when $1<b<\alpha+1$ and $K(n)<M$.  If $\alpha>1/2$, then
\[
    \frac1{\Neff}\sum_{i\le M}\min\{1,n^2\tmu_i^2\}\asymp\frac{\min\{M,K(n)\}}{\Neff}.
\]

\subsection{Compute-optimal derivation}
For fixed $N$, minimize $K^{1-b}+K/N$ under $K\le N^{1/\alpha}$.  The unconstrained optimum satisfies $K^{-b}\asymp N^{-1}$, hence $K\asymp N^{1/b}$.  If $b\le\alpha$, this exceeds the time cap, so $K\asymp N^{1/\alpha}$; otherwise $K\asymp N^{1/b}$.  Combining with $M=C/N$ and balancing $M^{1-b}$ against the optimized $N$ term yields Theorem~\ref{thm:compute}.

\subsection{Diagnostic consistency}
Suppose on a fitting window $i\in[L,U]$,
\[
    \lambda_i=c_\lambda i^{-a}e^{\varepsilon_i},\qquad
    \mu_i=c_\mu i^{-\alpha}e^{\delta_i},
\]
with $|\varepsilon_i|,|\delta_i|\le\eta$.  Least-squares log-log slope estimates satisfy
\[
    |\widehat a-a|+|\widehat\alpha-\alpha|\lesssim\eta/\log(U/L).
\]
Thus $\widehat q=1-\widehat\alpha/\widehat a$ is consistent whenever $a$ is bounded away from zero and the fitting window has nontrivial logarithmic width.

\subsection{Hidden-spectrum oracle separation}
If coordinatewise adaptivity has $q_{\rm coord}=0$ while a spectral oracle has $q_{\rm spec}=1/2$, then with $\beta(q)=(b-1)/(\max\{a(1-q),b\}+1)$,
\[
    \frac{R^\star_{q=0}(C)-\sigma^2}{R^\star_{q=1/2}(C)-\sigma^2}\asymp C^{\beta(1/2)-\beta(0)}.
\]
Scalar learning-rate tuning rescales the effective horizon by a constant and cannot change $\beta(q)$.
